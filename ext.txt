If you're currently using `multer.memoryStorage()`, you're already keeping the file in memory. To upload directly to Cloudinary instead of saving locally, you need to:

1. Configure Cloudinary.
2. Convert the uploaded buffer to a stream.
3. Upload the stream to Cloudinary.
4. Return the Cloudinary URL.

### 1. Install dependencies

```bash
npm install cloudinary
```

---

### 2. Cloudinary Config

**config/cloudinary.js**

```js
import { v2 as cloudinary } from "cloudinary";

cloudinary.config({
  cloud_name: process.env.CLOUDINARY_CLOUD_NAME,
  api_key: process.env.CLOUDINARY_API_KEY,
  api_secret: process.env.CLOUDINARY_API_SECRET,
});

export default cloudinary;
```

---

### 3. Environment Variables

```env
CLOUDINARY_CLOUD_NAME=your_cloud_name
CLOUDINARY_API_KEY=your_api_key
CLOUDINARY_API_SECRET=your_api_secret
```

---

### 4. Update Controller

**message.controller.js**

```js
import cloudinary from "../../config/cloudinary.js";
import streamifier from "streamifier";

export const uploadMedia = async (req, res) => {
  try {
    if (!req.file) {
      return res.status(400).json({
        success: false,
        message: "No file uploaded",
      });
    }

    const uploadFromBuffer = () =>
      new Promise((resolve, reject) => {
        const uploadStream = cloudinary.uploader.upload_stream(
          {
            folder: "chat-media",
            resource_type: "auto", // image, video, pdf, etc.
          },
          (error, result) => {
            if (error) reject(error);
            else resolve(result);
          }
        );

        streamifier.createReadStream(req.file.buffer).pipe(uploadStream);
      });

    const result = await uploadFromBuffer();

    return res.status(200).json({
      success: true,
      url: result.secure_url,
      publicId: result.public_id,
      resourceType: result.resource_type,
    });
  } catch (error) {
    console.error(error);

    return res.status(500).json({
      success: false,
      message: "Upload failed",
    });
  }
};
```

Install `streamifier`:

```bash
npm install streamifier
```

---

### 5. Router (No Major Changes)

```js
import { Router } from "express";
import { authenticate } from "../../middleware/auth.js";
import multer from "multer";
import { getMessages, uploadMedia } from "./message.controller.js";

const upload = multer({
  storage: multer.memoryStorage(),
  limits: {
    fileSize: 10 * 1024 * 1024,
  },
});

const router = Router();

router.get("/:chatId", authenticate, getMessages);
router.post(
  "/upload",
  authenticate,
  upload.single("file"),
  uploadMedia
);

export default router;
```

### Alternative (Recommended)

Instead of manually streaming, use `multer-storage-cloudinary`:

```bash
npm install multer-storage-cloudinary
```

Then Cloudinary becomes the Multer storage engine and files are uploaded directly during the request, eliminating the extra controller upload step. This is cleaner if **all uploaded files should always go to Cloudinary**.
